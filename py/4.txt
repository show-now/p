text = input("Enter a string: ")
text = text.lower()
vowels = "aeiou"
vowels_count = 0
consonants_count = 0

for word in text:
    if word.isalpha() and word in vowels:
        vowels_count += 1
    else:
        consonants_count += 1

print("Number of vowels:", vowels_count)
print("Number of consonants:", consonants_count)
