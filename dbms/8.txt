CREATE USER userA IDENTIFIED BY userA123;
CREATE USER userB IDENTIFIED BY userB123;

GRANT CREATE SESSION, CREATE TABLE, CREATE VIEW, CREATE PROCEDURE TO userA;
GRANT CREATE SESSION TO userB;

-- Give tablespace quota
ALTER USER userA QUOTA 10M ON USERS;

-- CONNECT AS userA
CREATE TABLE EMP1 (
  id INT,
  name VARCHAR(5)
);

GRANT SELECT, INSERT ON EMP1 TO userB;

-- CONNECT AS userB
DESC EMP1;
DESC userA.EMP1;

-- CONNECT AS userA
REVOKE SELECT, INSERT ON EMP1 FROM userB;

-- CONNECT AS userB
DESC userA.EMP1;
